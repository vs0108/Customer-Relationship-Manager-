		$(document).ready(function() {
			$("#add-customer").click(function() {
				var name = $("#name").val();
				var email = $("#email").val();
				var phone = $("#phone").val();
				if (name != "" && email != "" && phone != "") {
					$("#customers").append("<tr><td>" + name + "</td><td>" + email + "</td><td>" + phone + "</td></tr>");
					$("#name").val("");
					$("#email").val("");
					$("#phone").val("");
				}
				else {
					alert("Please fill in all fields.");
				}
			});
		});
	</script>
